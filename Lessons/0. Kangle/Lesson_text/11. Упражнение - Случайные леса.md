```python
# Код, который вы ранее использовали для загрузки данных
import pandas as pd
from sklearn.metrics import mean_absolute_error
from sklearn.model_selection import train_test_split
from sklearn.tree import DecisionTreeRegressor


# Путь к файлу для чтения
iowa_file_path = '../input/home-data-for-ml-course/train.csv'

home_data = pd.read_csv(iowa_file_path)
# Создайте целевой объект и назовите его y
y = home_data.SalePrice
# Создайте X
features = ['LotArea', 'YearBuilt', '1stFlrSF', '2ndFlrSF', 'FullBath', 'BedroomAbvGr', 'TotRmsAbvGrd']
X = home_data[features]

# Разделите на валидационные и тренировочные данные
train_X, val_X, train_y, val_y = train_test_split(X, y, random_state=1)

# Укажите модель
iowa_model = DecisionTreeRegressor(random_state=1)
# Обучите модель
iowa_model.fit(train_X, train_y)

# Сделайте валидационные предсказания и вычислите среднюю абсолютную ошибку
val_predictions = iowa_model.predict(val_X)
val_mae = mean_absolute_error(val_predictions, val_y)
print("Validation MAE when not specifying max_leaf_nodes: {:,.0f}".format(val_mae))

# Используя лучшее значение для max_leaf_nodes
iowa_model = DecisionTreeRegressor(max_leaf_nodes=100, random_state=1)
iowa_model.fit(train_X, train_y)
val_predictions = iowa_model.predict(val_X)
val_mae = mean_absolute_error(val_predictions, val_y)
print("Validation MAE for best value of max_leaf_nodes: {:,.0f}".format(val_mae))
```
